  "name": "string",
  "type": "Microsoft.DevTestLab/labs/virtualnetworks",
  "apiVersion": "2015-05-21-preview",
  "properties": {
    "allowedSubnets": [
      {
        "resourceId": "string",

        "allowPublicIp": "string"
      }
    ],
    "description": "string",
    "externalProviderResourceId": "string",
    "subnetOverrides": [
      {
        "resourceId": "string",

  "tags": {}
