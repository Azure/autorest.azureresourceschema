```json
{
  "name": "string",
  "type": "Microsoft.ApiManagement/service/templates",
  "apiVersion": "2017-03-01",
  "properties": {
    "subject": "string",
    "title": "string",
    "description": "string",
    "body": "string",
    "parameters": [
      {
        "name": "string",
        "title": "string",
```
